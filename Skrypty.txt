SKRYPTY GENERUJACE BAZY:

CREATE TABLE PRZEPROWADZONE_ZABIEGI(
ID_ZABIEGU INT(6) unsigned NOT NULL auto_increment primary key,
ID_LEKARZA INT(6) unsigned NOT NULL,
ID_PACJENTA int(6) unsigned NOT NULL,
ID_SALI int(6) unsigned NOT NULL,
ID_NARZADU int(6) unsigned NOT NULL,
DATA_ZABIEGU DATE NOT NULL,
PRZEBIEG TEXT,
CZY_OPERACJA_SIE_UDALA BIT NOT NULL,

 FOREIGN KEY (ID_LEKARZA) REFERENCES LEKARZE(ID_LEKARZA),
 FOREIGN KEY (ID_PACJENTA) REFERENCES LEKARZE(ID_PACJENTA),
 FOREIGN KEY (ID_SALI) REFERENCES LEKARZE(ID_SALI),
 FOREIGN KEY (ID_NARZADU) REFERENCES LEKARZE(ID_NARZADU)
);


CREATE TABLE LEKARZE(
ID_LEKARZA INT(6) unsigned NOT NULL auto_increment primary key,
IMIE VARCHAR(30) NOT NULL,
NAZWISKO VARCHAR(30) NOT NULL,
SPECJALIZACJA VARCHAR(70) NOT NULL
);

CREATE TABLE PACJENCI(
ID_PACJENTA INT(6) unsigned NOT NULL auto_increment primary key,
IMIE VARCHAR(30) NOT NULL,
NAZWISKO VARCHAR(30) NOT NULL,
GRUPA_KRWI VARCHAR(5) NOT NULL,
DATA_PRZYJECIA DATE,
CZY_JEST_DAWCA BIT NOT NULL,
CZY_JEST_W_SZPITALU BIT NOT NULL,
HISTORIA TEXT,
LEKARZ_PROWADZACY INT(6) unsigned NOT NULL,

 FOREIGN KEY (LEKARZ_PROWADZACY) REFERENCES LEKARZE(ID_LEKARZA)
);

CREATE TABLE MAGAZYN(
ID_NARZADU INT(6) unsigned NOT NULL auto_increment primary key,
ID_PACJENTA INT(6) unsigned NOT NULL,
NARZAD VARCHAR(50) NOT NULL,
DATA_POBRANIA DATE NOT NULL,

 FOREIGN KEY (ID_PACJENTA) REFERENCES PACJENCI(ID_PACJENTA)
);

CREATE TABLE SALE_OPERACYJNE(
ID_SALI INT(6) unsigned NOT NULL auto_increment primary key,
CZY_SALA_JEST_WOLNA BIT NOT NULL
);
